% SLIDE 3: HYPOTHÈSES DE CADRAGE DU PROJET
% ============================================================
\begin{frame}{Hypothèses de Cadrage du Projet}

\textbf{\large Hypothèses Méthodologiques}

\vspace{0.3cm}

\begin{itemize}
    \item \textbf{H1 - Périmètre:} Focus sur \textbf{8 moments de vie} à fort impact business
    \begin{itemize}
        \item Mariage/PACS, Naissance, Achat immobilier, Changement d'emploi, Retraite, Décès proche, Divorce, Création entreprise
    \end{itemize}
    
    \vspace{0.2cm}
    
    \item \textbf{H2 - Prédictibilité:} Signaux comportementaux détectables \textbf{3-6 mois avant} l'événement
    
    \vspace{0.2cm}
    
    \item \textbf{H3 - Données:} Données transactionnelles + CRM suffisantes pour modélisation
    
    \vspace{0.2cm}
    
    \item \textbf{H4 - ROI:} Rentabilité avec un taux de conversion de \textbf{15\%} sur offres ciblées
    
    \vspace{0.2cm}
    
    \item \textbf{H5 - Conformité:} Respect total du \textbf{RGPD} et éthique bancaire
\end{itemize}

\vspace{0.3cm}

\begin{block}{Approche de Validation}
Ateliers de validation avec les parties prenantes métier + Tests de faisabilité technique
\end{block}

\end{frame}

% ============================================================
% SLIDE 4: PHASE 1 - COMPRÉHENSION MÉTIER
% ============================================================
\begin{frame}{Phase 1: Compréhension Métier (Semaines 1-2)}

\textbf{\large Activités Clés}

\vspace{0.3cm}

\begin{enumerate}
    \item \textbf{Ateliers métier} (3-4 sessions)
    \begin{itemize}
        \item Directions: Marketing, Retail, Gestion de Patrimoine, Crédits
        \item Identification des 8 moments de vie prioritaires
        \item Compréhension des parcours clients actuels
    \end{itemize}
    
    \vspace{0.2cm}
    
    \item \textbf{Audit des données} (1 semaine)
    \begin{itemize}
        \item Données CRM, transactionnelles, comportementales, externes
        \item Évaluation de la qualité et complétude (scoring RGPD)
        \item Identification des gaps et enrichissements nécessaires
    \end{itemize}
    
    \vspace{0.2cm}
    
    \item \textbf{Définition des objectifs quantifiables}
    \begin{itemize}
        \item KPIs: Précision, Recall, F1-Score, Lift vs. baseline
        \item Horizon de prédiction: 3-6-12 mois
        \item ROI attendu: taux de conversion, satisfaction client
    \end{itemize}
\end{enumerate}

\end{frame}
