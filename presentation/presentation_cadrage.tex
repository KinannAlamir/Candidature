% SLIDE 3: PHASE 1 - COMPRÉHENSION MÉTIER
% ============================================================
\begin{frame}{Phase 1: Compréhension Métier (Semaines 1-2)}

\textbf{\large Activités Clés}

\vspace{0.3cm}

\begin{enumerate}
    \item \textbf{Ateliers métier} (3-4 sessions)
    \begin{itemize}
        \item Directions: Marketing, Retail, Gestion de Patrimoine, Crédits
        \item Identification des 8 moments de vie prioritaires
        \item Compréhension des parcours clients actuels
    \end{itemize}
    
    \vspace{0.2cm}
    
    \item \textbf{Audit des données} (1 semaine)
    \begin{itemize}
        \item Données CRM, transactionnelles, comportementales, externes
        \item Évaluation de la qualité et complétude (scoring RGPD)
        \item Identification des gaps et enrichissements nécessaires
    \end{itemize}
    
    \vspace{0.2cm}
    
    \item \textbf{Définition des objectifs quantifiables}
    \begin{itemize}
        \item KPIs: Précision, Recall, F1-Score, Lift vs. baseline
        \item Horizon de prédiction: 3-6-12 mois
        \item ROI attendu: taux de conversion, satisfaction client
    \end{itemize}
\end{enumerate}

\end{frame}

% ============================================================
% SLIDE 4: MOMENTS DE VIE & HYPOTHÈSES
% ============================================================
\begin{frame}{Moments de Vie Identifiés \& Hypothèses de Travail}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{\textcolor{wedRBlue}{8 Moments de Vie Prioritaires}}
\begin{enumerate}
    \item Mariage / PACS
    \item Naissance / Adoption
    \item Achat immobilier
    \item Changement d'emploi
    \item Retraite
    \item Décès d'un proche
    \item Divorce / Séparation
    \item Création d'entreprise
\end{enumerate}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{\textcolor{wedRGreen}{Hypothèses Clés}}
\begin{itemize}
    \item \textbf{H1:} Signaux comportementaux 3-6 mois avant l'événement
    \item \textbf{H2:} Données transactionnelles + CRM suffisantes
    \item \textbf{H3:} Précision >70\% pour top 3 événements
    \item \textbf{H4:} ROI positif avec 15\% de conversion
    \item \textbf{H5:} Conformité RGPD totale
\end{itemize}
\end{column}
\end{columns}

\vspace{0.4cm}

\begin{block}{Dataset Synthétique Créé}
\textbf{10,000 clients} - 40+ features (démographiques, financières, comportementales) - Horizon 12 mois
\end{block}

\end{frame}
